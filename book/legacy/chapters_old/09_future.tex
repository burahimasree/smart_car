\chapter{Future Scope}

\section{Hardware Upgrades}
\begin{itemize}
    \item \textbf{Accelerator}: Adding a Google Coral USB Stick would jump Vision FPS from 3.5 to 30+.
    \item \textbf{Lidar}: Replacing Sonar with a cheap 2D Lidar (RPLidar A1) for SLAM mapping.
\end{itemize}

\section{Software Roadmap}
\begin{itemize}
    \item \textbf{Local LLM}: Migrating from Cloud Gemini to a localized TinyLlama quantized model (requires Pi 5 or Jetson).
    \item \textbf{Web Dashboard}: A React-based telemetry screen hosted on Nginx.
\end{itemize}
